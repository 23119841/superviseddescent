145  template<class H, class OnTrainingEpochCallback>
146  void train(cv::Mat x, cv::Mat x0, cv::Mat y, H h, OnTrainingEpochCallback onTrainingEpochCallback)
147  {
148  using cv::Mat;
149  Mat currentX = x0;
150  for (size_t regressorLevel = 0; regressorLevel < regressors.size(); ++regressorLevel) {
151  // 1) Project current parameters x to feature space:
152  // Enqueue all tasks in a thread pool:
153  auto concurentThreadsSupported = std::thread::hardware_concurrency();
154  if (concurentThreadsSupported == 0) {
155  concurentThreadsSupported = 4;
156  }
157  utils::ThreadPool threadPool(concurentThreadsSupported);
158  std::vector<std::future<typename std::result_of<H(Mat, size_t, int)>::type>> results; // will be float or Mat. I might remove float for the sake of code clarity, as it's only useful for very simple examples.
159  results.reserve(currentX.rows);
160  for (int sampleIndex = 0; sampleIndex < currentX.rows; ++sampleIndex) {
161  results.emplace_back(
162  threadPool.enqueue(h, currentX.row(sampleIndex), regressorLevel, sampleIndex)
163  );
164  }
165  // Gather the results from all threads and store the features:
166  Mat features;
167  for (auto&& result : results) {
168  features.push_back(result.get());
169  }
170  // Set the observed values, depending on if a template y is used:
171  Mat observedValues;
172  if (y.empty()) { // unknown y training case
173  observedValues = features;
174  }
175  else { // known y
176  observedValues = features - y;
177  }
178  Mat b = currentX - x;
179  // 2) Learn using that data:
180  regressors[regressorLevel].learn(observedValues, b);
181  // 3) Apply the learned regressor and use the predictions to learn the next regressor in next loop iteration:
182  Mat x_k; // x_k = currentX - R * (h(currentX) - y):
183  for (int sampleIndex = 0; sampleIndex < currentX.rows; ++sampleIndex) {
184  // No need to re-extract the features, we already did so in step 1)
185  x_k.push_back(Mat(currentX.row(sampleIndex) - regressors[regressorLevel].predict(observedValues.row(sampleIndex))));
186  }
187  currentX = x_k;
188  onTrainingEpochCallback(currentX);
189  }
190  };
