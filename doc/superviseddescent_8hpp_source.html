143  template<class H, class OnTrainingEpochCallback>
144  void train(cv::Mat x, cv::Mat x0, cv::Mat y, H h, OnTrainingEpochCallback onTrainingEpochCallback)
145  {
146  using cv::Mat;
147  Mat currentX = x0;
148  for (size_t regressorLevel = 0; regressorLevel < regressors.size(); ++regressorLevel) {
149  // 1) Project current parameters x to feature space:
150  Mat features;
151  for (int sampleIndex = 0; sampleIndex < currentX.rows; ++sampleIndex) {
152  features.push_back(h(currentX.row(sampleIndex), regressorLevel, sampleIndex));
153  }
154  Mat observedValues;
155  if (y.empty()) { // unknown y training case
156  observedValues = features;
157  }
158  else { // known y
159  observedValues = features - y;
160  }
161  Mat b = currentX - x;
162  // 2) Learn using that data:
163  regressors[regressorLevel].learn(observedValues, b);
164  // 3) Apply the learned regressor and use the predictions to learn the next regressor in next loop iteration:
165  Mat x_k; // x_k = currentX - R * (h(currentX) - y):
166  for (int sampleIndex = 0; sampleIndex < currentX.rows; ++sampleIndex) {
167  // No need to re-extract the features, we already did so in step 1)
168  x_k.push_back(Mat(currentX.row(sampleIndex) - regressors[regressorLevel].predict(observedValues.row(sampleIndex))));
169  }
170  currentX = x_k;
171  onTrainingEpochCallback(currentX);
172  }
173  };

259  template<class H>
260  cv::Mat predict(cv::Mat x0, cv::Mat y, H h)
261  {
262  using cv::Mat;
263  Mat currentX = x0;
264  for (size_t r = 0; r < regressors.size(); ++r) {
265  // calculate x_k = currentX - R * (h(currentX) - y):
266  Mat observedValues;
267  if (y.empty()) { // unknown y training case
268  observedValues = h(currentX, r);
269  }
270  else { // known y
271  observedValues = h(currentX, r) - y;
272  }
273  Mat x_k = currentX - regressors[r].predict(observedValues);
274  currentX = x_k;
275  }
276  return currentX;
