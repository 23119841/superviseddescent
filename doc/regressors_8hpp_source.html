131  cv::Mat getMatrix(cv::Mat data, int numTrainingElements)
132  {
133  switch (regularisationType)
134  {
135  case RegularisationType::Manual:
136  // We just take lambda as it was given, no calculation necessary.
137  break;
138  case RegularisationType::MatrixNorm:
139  // The given lambda is the factor we have to multiply the automatic value with:
140  lambda = lambda * static_cast<float>(cv::norm(data)) / static_cast<float>(numTrainingElements);
141  break;
142  default:
143  break;
144  }
145 
146  cv::Mat regulariser = cv::Mat::eye(data.rows, data.cols, CV_32FC1) * lambda;
147 
148  if (!regulariseLastRow) {
149  // no lambda for the bias:
150  regulariser.at<float>(regulariser.rows - 1, regulariser.cols - 1) = 0.0f;
151  }
152  return regulariser;
153  };

209  bool learn(cv::Mat data, cv::Mat labels) override
210  {
211  using cv::Mat;
212  // Map the cv::Mat data and labels to Eigen matrices:
213  Eigen::Map<Eigen::Matrix<float, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>> A_Eigen(data.ptr<float>(), data.rows, data.cols);
214  Eigen::Map<Eigen::Matrix<float, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>> labels_Eigen(labels.ptr<float>(), labels.rows, labels.cols);
215 
216  Eigen::Matrix<float, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> AtA_Eigen = A_Eigen.transpose() * A_Eigen;
217 
218  // Note: This is a bit of unnecessary back-and-forth mapping, just for the regularisation:
219  Mat AtA_Map(static_cast<int>(AtA_Eigen.rows()), static_cast<int>(AtA_Eigen.cols()), CV_32FC1, AtA_Eigen.data());
220  Mat regularisationMatrix = regulariser.getMatrix(AtA_Map, data.rows);
221  Eigen::Map<Eigen::Matrix<float, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>> reg_Eigen(regularisationMatrix.ptr<float>(), regularisationMatrix.rows, regularisationMatrix.cols);
222 
223  AtA_Eigen = AtA_Eigen + reg_Eigen;
224 
225  // Perform a ColPivHouseholderQR (faster than FullPivLU) that allows to check for invertibility:
226  Eigen::ColPivHouseholderQR<Eigen::Matrix<float, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>> qrOfAtA(AtA_Eigen);
227  auto rankOfAtA = qrOfAtA.rank();
228  if (!qrOfAtA.isInvertible()) {
229  // Eigen may return garbage (their docu is not very specific). Best option is to increase regularisation.
230  std::cout << "The regularised AtA is not invertible. We continued learning, but Eigen may return garbage (their docu is not very specific). (The rank is " << std::to_string(rankOfAtA) << ", full rank would be " << std::to_string(AtA_Eigen.rows()) << "). Increase lambda." << std::endl;
231  }
232  Eigen::Matrix<float, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> AtAInv_Eigen = qrOfAtA.inverse();
233 
234  // x = (AtAReg)^-1 * At * b:
235  Eigen::Matrix<float, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> x_Eigen = AtAInv_Eigen * A_Eigen.transpose() * labels_Eigen;
236 
237  // Map the resulting x back to a cv::Mat by creating a Mat header:
238  Mat x(static_cast<int>(x_Eigen.rows()), static_cast<int>(x_Eigen.cols()), CV_32FC1, x_Eigen.data());
239 
240  // We have to copy the data because the underlying data is managed by Eigen::Matrix x_Eigen, which will go out of scope after we leave this function:
241  this->x = x.clone();
242 
243  return qrOfAtA.isInvertible();
244  };
